AP1: ¿Cuáles son los estacionamientos disponibles y son en la vía pública o fuera de ella? 
PREFIX edintapar: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/aparcamiento#>

SELECT ?parking ?tipo
WHERE {
  {
    ?parking a edintapar:OnStreetParking .
    BIND("En la vía pública" AS ?tipo)
  }
  UNION
  {
    ?parking a edintapar:OffStreetParking .
    BIND("Fuera de la vía pública" AS ?tipo)
  }
}
ORDER BY ?parking
Con nombre/etiqueta 
PREFIX esintinfp: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/infraestructurapublica#>
PREFIX rdfs:      <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?parking ?label ?tipo
WHERE {
  {
    ?parking a esintinfp:OnStreetParking .
    OPTIONAL { ?parking rdfs:label ?label }
    BIND("En vía pública" AS ?tipo)
  }
  UNION
  {
    ?parking a esintinfp:OffStreetParking .
    OPTIONAL { ?parking rdfs:label ?label }
    BIND("Fuera de vía pública" AS ?tipo)
  }
}
ORDER BY LCASE(STR(?label)) ?parking

Fichero: AP03.sparql
AP3: Para un aparcamiento, ¿qué grupos de plazas existen y para qué tipo de vehículo están reservadas (ej. eléctricos, motos, PMR)? 
PREFIX edintapar: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/aparcamiento#>
PREFIX edintveh: <http://vocab.linkeddata.es/datosabiertos/def/transporte/censovehiculos#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?parking ?grupo ?vehicleType ?vehicleTypeLabel ?numSpots
WHERE {
  VALUES ?parking { <URI_DEL_APARCAMIENTO> }

  # Grupos de plazas por tipo de vehículo
  ?parking edintapar:hasSpotsByVehicleType ?grupo .

  # Tipo de vehículo (SKOS Concept)
  ?grupo edintveh:vehicleType ?vehicleType .

  # Número de plazas
  OPTIONAL { ?grupo edintapar:numberOfParkingSpots ?numSpots . }

  # Etiqueta del tipo de vehículo (si existe). Algo análogo para resto de etiquetas si se quiere
  OPTIONAL { ?vehicleType rdfs:label ?vehicleTypeLabel . }
}
ORDER BY ?vehicleTypeLabel
Para todos los aparcamientos 
PREFIX edintapar: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/aparcamiento#>
PREFIX edintveh: <http://vocab.linkeddata.es/datosabiertos/def/transporte/censovehiculos#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?parking ?grupo ?vehicleType ?vehicleTypeLabel ?numSpots
WHERE {
  ?parking a edintapar:Parking ;
           edintapar:hasSpotsByVehicleType ?grupo .

  ?grupo edintveh:vehicleType ?vehicleType .

  OPTIONAL { ?grupo edintapar:numberOfParkingSpots ?numSpots . }
  OPTIONAL { ?vehicleType rdfs:label ?vehicleTypeLabel . }
}
ORDER BY ?parking ?vehicleTypeLabel
Y por tipo de vehículo 
PREFIX esintinfp: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructurapublica#>
PREFIX skos:      <http://www.w3.org/2004/02/skos/core#>
PREFIX rdfs:      <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?parking ?parkingLabel
       ?vehicleType ?vehicleTypeLabel
       (GROUP_CONCAT(DISTINCT ?groupLabel; SEPARATOR=", ") AS ?grupos)	# Agrupados por tipo de vehículo
       (SUM(?numPlazas) AS ?totalPlazas)
WHERE {
  ?parking a esintinfp:Parking ;		# Para todos los aparcamientos
           esintinfp:hasSpotsByVehicleType ?group .

  OPTIONAL { ?parking rdfs:label ?parkingLabel }
  OPTIONAL { ?group rdfs:label ?groupLabel }

  ?group esintinfp:hasVehicleTypeForSpots ?vehicleType .
  OPTIONAL { ?vehicleType skos:prefLabel ?vehicleTypeLabel }

  OPTIONAL { ?group esintinfp:numberOfSpots ?numPlazas }
}
GROUP BY ?parking ?parkingLabel ?vehicleType ?vehicleTypeLabel
ORDER BY ?parkingLabel ?vehicleTypeLabel
Fichero: AP04.sparql
AP4: ¿Qué tipo de permiso se requiere para aparcar en un grupo de plazas (ej. residentes, carga y descarga) de un aparcamiento? 
PREFIX edintapar: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/aparcamiento#>
PREFIX edintveh: <http://vocab.linkeddata.es/datosabiertos/def/transporte/censovehiculos#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>

SELECT ?parking ?group ?permitType ?permitLabel ?numSpots
WHERE {
  VALUES ?parking { <URI_DEL_APARCAMIENTO> }

  # Grupo de plazas por tipo de permiso
  ?parking edintapar:hasSpotsByParkingPermitType ?group .

  # Tipo de permiso (SKOS Concept)
  ?group edintveh:permitType ?permitType .

  # Número de plazas en ese grupo
  OPTIONAL { ?group edintapar:numberOfParkingSpots ?numSpots . }

  # Etiqueta legible del tipo de permiso
  OPTIONAL { ?permitType skos:prefLabel|rdfs:label ?permitLabel . }
}
ORDER BY ?permitLabel
Para todos los grupos de plazas 
PREFIX esintinfp: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructurapublica#>
PREFIX esinfpub:  <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/infraestructuraPublica#>
PREFIX skos:      <http://www.w3.org/2004/02/skos/core#>
PREFIX rdfs:      <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?group ?groupLabel ?permitType ?permitTypeLabel
WHERE {
  ?group a esinfpub:ParkingSpotsByPermitType ;		
         esintinfp:hasPermitTypeForSpots ?permitType .

  OPTIONAL { ?group rdfs:label ?groupLabel }
  OPTIONAL { ?permitType skos:prefLabel ?permitTypeLabel }
}
ORDER BY ?groupLabel ?permitTypeLabel

AP5: ¿Cuáles son los aparcamientos existentes en una zona regulada concreta? 
PREFIX edintapar: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/aparcamiento#>

SELECT ?parking
WHERE {
  ?parking a edintapar:OnStreetParking ;
           edintapar:isPartOfZone <URI_DE_LA_ZONA_REGULADA> .
}
ORDER BY ?parking
Con nombre/etiqueta de la zona regulada 
PREFIX edintapar: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/aparcamiento#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?parking ?zoneLabel
WHERE {
  ?parking a edintapar:OnStreetParking ;
           edintapar:isPartOfZone ?zone .

  FILTER (?zone = <URI_DE_LA_ZONA_REGULADA>)

  OPTIONAL { ?zone rdfs:label ?zoneLabel . }
}
ORDER BY ?parking
